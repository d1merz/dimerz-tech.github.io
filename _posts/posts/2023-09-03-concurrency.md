    type Output;

    Ready(T),

}

–û—Å–Ω–æ–≤–Ω–∞—è —Å—Ç—Ä—É–∫—Ç—É—Ä–∞ –¥–ª—è –∫–æ—Ç–æ—Ä–æ–π –º—ã –Ω–∞–ø–∏—à–µ–º —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—é —Ç—Ä–µ–π—Ç–∞ Future

    time: SystemTime
}

/*
–†–µ–∞–ª–∏–∑—É–µ–º —Ç—Ä–µ–π—Ç Future –¥–ª—è —Å—Ç—Ä—É–∫—Ç—É—Ä—ã Sleep. –§—É–Ω–∫—Ü–∏—è poll –±—É–¥–µ—Ç –≤–æ–∑–≤—Ä–∞—â–∞—Ç—å Ready,

impl CustomFuture for Sleep {
    type Output = ();
    fn poll(&mut self, _wake: fn()) -> Poll<Self::Output> {
        let sec = self.time.elapsed().unwrap().as_secs();
        if sec % 5 == 0 {
            Poll::Ready(())
        } else {
            Poll::Pending
        }
    }
}

impl Sleep {
    pub fn new() -> Self {
        Sleep {time: SystemTime::now()}
    }
  /*
  –î–∞–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –±—É–¥–µ—Ç —Å–æ–æ–±—â–∞—Ç—å, –≥–æ—Ç–æ–≤–∞ –¥–ª—è Future –∏–ª–∏ –Ω–µ—Ç
   */
    pub fn is_ready(&mut self) -> bool {
        match self.poll(||{}) {
            Poll::Ready(()) => true,
            Poll::Pending => false
        }
    }
}

  /*
  –ó–∞–ø—É—Å–∫–∞–µ–º –æ—Ç–¥–µ–ª—å–Ω—ã–π –ø–æ—Ç–æ–∫, –∫–æ—Ç–æ—Ä—ã–π –±—É–¥–µ—Ç –æ–ø—Ä–∞—à–∏–≤–∞—Ç—å Future,

        let mut future = Sleep::new();
        loop {
            if !future.is_ready() {
                println!("Yielded");
                sleep(Duration::from_secs(1));
            } else {

            }
        }
    });
    loop {
        println!("It's a main thread!");

    }
}

#### Tasks
